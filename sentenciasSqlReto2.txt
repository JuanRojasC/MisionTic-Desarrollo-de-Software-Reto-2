
1. CREACION DE TABLAS

Create table CLIENT(
    ID number PRIMARY KEY,
    Name VARCHAR(4000),
    Email VARCHAR2(20),
    Age NUMBER
);

Create table MESSAGE(
    ID NUMBER PRIMARY KEY,
    Messagetext VARCHAR2(4000) 
);

2. MANEJADORES TABLA CLIENT
GET
select * from client

POST
BEGIN
    INSERT INTO CLIENT
    (ID, NAME, EMAIL, AGE)
    VALUES (:id,:name,:email,:age);
    :status_code:= 201;
    END;

PUT
BEGIN
    UPDATE CLIENT 
    SET ID =: id, NAME =: name, EMAIL =: email, AGE =: age
    WHERE ID =: id;
    :status_code:=201;
    END;

DELETE
BEGIN
    DELETE FROM CLIENT
    WHERE ID=:id;
    :status_code:=204;
    END;

3. MANEJADORES TABLA MESSAGE
GET
select * from message

POST
BEGIN
    INSERT INTO MESSAGE
    (ID, MESSAGETEXT)
    VALUES (:id, :messagetext);
    : status_code := 201;
